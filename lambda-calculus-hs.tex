
\begin{frame}[fragile]
\begin{minted}{haskell}
data Term a = 
  -- for lambda calculus
    TmVar a 
  | TmLam (Scope () Term a) 
  | TmApp (Term a) (Term a)
  -- for examples
  | TmInt Int
  | TmAdd (Term a) (Term a)
  deriving (Eq, Ord, Show, Functor, Foldable, Traversable)

instance Monad Term where
  return = TmVar

  TmVar x >>= f = f x
  TmLam s >>= f = TmLam (s >>>= f)
  TmApp x y >>= f = TmApp (x >>= f) (y >>= f)
  TmInt i >>= _ = TmInt i
  TmAdd x y >>= f = TmAdd (x >>= f) (y >>= f)
\end{minted}
\end{frame}

\begin{frame}[fragile]
\begin{minted}{haskell}
instance Eq1 Term where ...
instance Ord1 Term where ...
instance Show1 Term where ...

instance Applicative Term where
  pure = return
  (<*>) = ap
\end{minted}
\end{frame}

\begin{frame}[fragile]
\begin{minted}{haskell}
tmVar :: a -> Term a
tmVar = TmVar

tmApp :: Term a -> Term a -> Term a
tmApp = TmApp

tmInt :: Int -> Term a
tmInt = TmInt

tmAdd :: Term a -> Term a -> Term a
tmAdd = TmAdd
\end{minted}
\end{frame}

\begin{frame}[fragile]
\begin{minted}{haskell}
tmLam :: Eq a => a -> Term a -> Term a
tmLam x t = TmLam (abstract1 x t)

reduceAppLam :: Term a -> Maybe (Term a)
reduceAppLam (TmApp (TmLam s) tm) = 
  Just $ instanitate1 tm s
reduceAppLam _ = 
  Nothing
\end{minted}
\end{frame}

\begin{frame}[fragile]
\begin{minted}{haskell}
-- x
> tmVar "x"
TmVar "x"
\end{minted}
\end{frame}

\begin{frame}[fragile]
\begin{minted}{haskell}
-- add = x + 2
> let add = tmAdd (tmVar "x") (tmInt 2)

TmAdd 
  (TmVar "x") 
  (TmInt 2)
\end{minted}
\end{frame}

\begin{frame}[fragile]
\begin{minted}{haskell}
-- add = x + 2

-- fn = \x . x + 2
> let fn = tmLam "x" add
> fn
TmLam (Scope 
  (TmAdd 
    (TmVar (B ())) 
    (TmInt 2)))
\end{minted}
\end{frame}

\begin{frame}[fragile]

> let res = tmApp fn (tmInt 1)
> res
TmApp (TmLam (Scope 
        (TmAdd 
          (TmVar (B ())) 
          (TmInt 2)))) 
      (TmInt 1)
\end{minted}
\end{frame}

\begin{frame}[fragile]
\begin{minted}{haskell}
-- res = \x . x + 2 @ 1
> res
TmApp (TmLam (Scope 
        (TmAdd 
          (TmVar (B ())) 
          (TmInt 2)))) 
      (TmInt 1)

> step res
Just $ TmAdd 
         (TmInt 1) 
         (TmInt 2)

> step res >>= step
Just $ TmInt 3
\end{minted}
\end{frame}
