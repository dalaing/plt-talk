
\section{Simply Typed Lambda Calculus}

\begin{frame}
\begin{displaymath}
    \begin{aligned}
t \quad:=\quad& ~ \ldots &\\
  & ~ x \quad\quad &variable\\
  & ~ \lambda~x \highlight{{:} T} . t \quad\quad &abstraction\\
  & ~ t ~ t \quad\quad &function~application\\
v \quad:=\quad& ~ \ldots &\\
  & ~ \lambda~x \highlight{{:} T} . t \quad\quad &abstraction\\
T \quad:=\quad& ~ \ldots &\\
  & ~ \highlight{T \rightarrow T} \quad\quad &function~arrow\\
    \end{aligned}
\end{displaymath}
\end{frame}

\begin{frame}
  \infrule[E-App1]
  {t_1 \longrightarrow {t_1}^{\prime}}
  {t_1~t_2 \longrightarrow {t_1}^{\prime}~t_2}
  \infrule[E-App2]
  {t_2 \longrightarrow {t_2}^{\prime}}
  {v_1~t_2 \longrightarrow v_1~{t_2}^{\prime}}
  \infrule[E-LamApp]
  {}
  {(\lambda~x \highlight{{:} T} . t_1) t_2 \longrightarrow \left[x \mapsto t_2 \right]t_1}
\end{frame}

\begin{frame}
  \infrule[T-Var]
  {x {:} T \in \Gamma}
  {\Gamma \vdash x {:} T}

  \infrule[T-App]
  {\Gamma \vdash t_1 {:} T_1 \rightarrow T_2 \andalso \Gamma \vdash t_2 {:} T1}
  {\Gamma \vdash t_1~t_2 {:} T_2}

  \infrule[T-Lam]
  {\Gamma , x {:} T_1 \vdash T_2}
  {\Gamma \vdash \left( \lambda~x {:} T_1 . t \right) {:} T_1 \rightarrow T_2}
\end{frame}
